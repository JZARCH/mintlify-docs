title: 'Create Fund Wallet Request'
api: 'POST /wallet/fund/requests/create'

Allowed TransferType : `domestic-wire` ,`international-wire`

## Authorization

<ParamField header="Authorization" type="string" required>
  Bearer token. Format: `Bearer {{AUTH_TOKEN}}`.
</ParamField>

<ParamField body="depositorName" type="string">

</ParamField>

<ParamField body="toWallet" type="string">
  Request body field `toWallet`.
</ParamField>

<ParamField body="amount" type="number">
  Request body field `amount`.
</ParamField>

<ParamField body="description" type="string">
  Request body field `description`.
</ParamField>

<ParamField body="transferType" type="string">
  Request body field `transferType`.
</ParamField>

## Request

Endpoint: `POST /wallet/fund/requests/create`

### Example

```bash
curl --location "$BASE_URL/wallet/fund/requests/create" \
  --header "Authorization: Bearer $AUTH_TOKEN" \
  --header "Accept: application/json"
```
